DECLARE @TABLE_NAME VARCHAR(300)
SET @TABLE_NAME='Customer' --輸入TABLE名稱
SELECT COLUMN_NAME, IS_NULLABLE,DATA_TYPE,CHARACTER_MAXIMUM_LENGTH,CHARACTER_OCTET_LENGTH,ORDINAL_POSITION
,DATA_TYPE+'('+convert(varchar,CHARACTER_MAXIMUM_LENGTH)+')' as uniontype
FROM information_schema.COLUMNS 
where table_name=@TABLE_NAME
order by ORDINAL_POSITION
